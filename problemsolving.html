1.Find the Smaller Angle

 var Minimal_Angle = (h, m) => 
                {
                     var hourAngle = ((h * 60) + m) * 0.5;
                     var minuteAngle = m * 6;
                     var angle = Math.abs(hourAngle - minuteAngle)
                       if(angle > 180){
                            return 360 - angle
                       }
                      else{
                            return angle
                       }
                };

3.Perfect Number Check.

 var Perfect_Check = (N) => 
                 {   
                     var sum = 0;
                     for(let i = 1 ; i <= N; i++){ 
                         if(N % i == 0){
                            sum = sum + i;
                         }
                      }
            
                      if(sum == N){
                         return "YES"
                      }
            
                     else{
                         return "NO"
                     }
                
          };
